let redux = require("redux");
let createStore = redux.legacy_createStore;

// action type
let ADD_HERO = "ADD_HERO";
let REMOVE_HERO = "REMOVE_HERO";
let ADD_MOVIE = "ADD_MOVIE";
let REMOVE_MOVIE = "REMOVE_MOVIE";

// action creator
let addHero = () => {
    return {
        type : ADD_HERO
    }
};
let removeHero = () => {
    return {
        type : REMOVE_HERO
    }
};
let addMovie = () => {
    return {
        type : ADD_MOVIE
    }
};
let removeMovie = () => {
    return {
        type : REMOVE_MOVIE
    }
};

// initial State
let initalState = {
    numberOfHeroes : 0,
    numberOfMovies : 0
}

// reducer are purefunctions 
let reducer = (state = initalState, action) => {
    switch(action.type){
        case ADD_HERO : return {...state, numberOfHeroes : state.numberOfHeroes + 1 }
        case REMOVE_HERO : return {...state, numberOfHeroes : state.numberOfHeroes - 1 }
        case ADD_MOVIE : return {...state, numberOfMovies : state.numberOfMovies + 1 }
        case REMOVE_MOVIE : return {...state, numberOfMovies : state.numberOfMovies - 1 }
        default : return state
    }
}

let store = createStore(reducer);
// initial state in the store
console.log(store.getState());

store.subscribe(() => {
    console.log(store.getState());
});


store.dispatch(addHero());
store.dispatch(addHero());
store.dispatch(addHero());
store.dispatch(removeHero());
store.dispatch(removeHero());
store.dispatch(addMovie());
store.dispatch(addMovie());
store.dispatch(addMovie());
store.dispatch(removeMovie());
store.dispatch(removeMovie());
